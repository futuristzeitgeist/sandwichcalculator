<body>

    <div id="numberContainer"></div>

    <p id="sandwichInfo"></p>

    <script>

        const originalNumbers = [1, 20, 11.4, 9]; // Initial numbers for Loaves of bread, Peanut butter (oz), Jelly (oz), Number of sandwiches

        const labels = ["Loaves of bread", "Peanut butter (oz)", "Jelly (oz)", "Number of sandwiches"];

        let inputs = [];


        function setup() {

            const container = document.getElementById('numberContainer');

            originalNumbers.forEach((number, index) => {

                const input = document.createElement('input');

                input.type = 'number';

                input.value = formatNumber(number);

                input.onchange = () => handleInputChange(index);


                const label = document.createElement('label');

                label.textContent = labels[index] + ": ";

                label.appendChild(input);


                const paragraph = document.createElement('p');

                paragraph.appendChild(label);

                container.appendChild(paragraph);


                inputs.push(input);

            });

            updateSandwichInfo();

        }


        function handleInputChange(changedIndex) {

            const changedValue = parseFloat(inputs[changedIndex].value);

            const originalValue = originalNumbers[changedIndex];

            const ratio = changedValue / originalValue;


            originalNumbers.forEach((originalNum, index) => {

                if (index !== changedIndex) {

                    const newValue = originalNum * ratio;

                    inputs[index].value = formatNumber(newValue); // Format number to remove unnecessary decimals

                }

            });

            updateSandwichInfo();

        }


        function updateSandwichInfo() {

            const numSandwiches = parseFloat(inputs[3].value);

            const timeNeeded = formatNumber(numSandwiches * 1.71);

            const friendPower = Math.max(numSandwiches / 70, 1);

            const friendText = friendPower === 1 ? "friend" : "friends"; // Check if friendPower is 1 to decide between 'friend' and 'friends'


            const info = `To make ${numSandwiches} sandwiches, one friend will need on average ${timeNeeded} minutes. ` +

            

                         `For an average sandwich-making party of 2 hours, to create ${numSandwiches} sandwiches, ` +

                         `you will need ${formatNumber(friendPower)} ${friendText} total.`;

            document.getElementById('sandwichInfo').textContent = info;

        }


        function formatNumber(num) {

            return num % 1 === 0 ? num.toString() : num.toFixed(2);

        }


        setup();

    </script>
